<div class="col-md-10 col-xs-12 detailNews">
<span class="show-define-area"></span>
<p>PT Kereta Commuter Indonesia (KCI) will provide passengers with free <em>takjil</em> (breaking-of-the-fast snacks) from May 15 to 17 at 6&nbsp;stations, namely Jakarta Kota, Gondangdia, Cikini, Sudirman and Tanah Abang.</p>
<p>&ldquo;To enliven this year&rsquo;s Ramadan, KCI will distribute free <em>takjil</em> at 6&nbsp;stations alternately from May 15 to 17,&rdquo; KCI spokesperson Anne Purba said on Tuesday as quoted by&nbsp;<em><a href="http://kompas.com/">kompas.com</a></em>.</p>
<p>The <em>takjil</em> will be a packet of light snack and a beverage to break the passengers&rsquo; fast. The passengers are also allowed to eat or drink <em>takjil</em> on the train, from sunset or&nbsp;<em>maghrib</em>&nbsp;until one hour afterward.</p>
<p>On the other days, passengers are also allowed to break their fast on the train, but not with food that emanates strong smell. (vla)</p>

    function load_more(e) {
        stillRunning = true;
        $.ajax({
            url: '/more-article',
            type: "GET",
            datatype: "json",
            data: {
                article: e,
                count: $(".loop").length
            },
            success: function(jsondata) {
                if (jsondata.status == true) {
                    $("#more_content").append(jsondata.data);
                    $( jsondata.data ).insertAfter( $( ".loop:last" ) );
                    $(".loop:last").attr('data-href', '/' + jsondata.url);
                    $(".loop:last").attr('data-title', jsondata.title);
                    stillRunning = false;
                    getShareCount('https://' + window.location.hostname + '/' + jsondata.url, jsondata.title);
                    $('.loop:last .addToBookmark').attr('data-url',jsondata.url).attr('data-title', jsondata.title).attr('data-date', jsondata.date);
                    isBookmarked($(".loop").length - 1, jsondata.url);
                    moreContent();
                }
                tabMultiple($(".loop").length - 1);
                bookmarkSidebar($(".loop").length - 1, 'uND8tQeb4zXRDwRGK8oZmnEP0zKw3HiiUDpLE6al');
                defineKeyword();
            }
        });
    }

    function check() {
            $.each($('.loop'), function(i, e){
                var r = $(this).offset();

                if ($(window).scrollTop() > r.top  - ($(window).height() * 1/2 ) &&
                    $(window).scrollTop() < r.top + $(this).height() - ($(window).height() * 1/2 )
                   )
                {
                    if (articleOnScreen != i) {
                        articleOnScreen = i;

                        var newTitle = $(this).attr('data-title');
                        if (newTitle === undefined) {
                            newTitle = document.getElementsByTagName("title")[0].innerHTML;
                        }
                        window.document.title = newTitle;

                        var url = 'https://' + window.location.hostname + $(this).attr('data-href');
                        window.history.pushState("object or string", "Title", url);
                        console.log('ga fired, page: ' + url);
                        ga('set', 'page', url);
                        ga('send', 'pageview');
                    }                
                }
            });
    }

    function isBookmarked(i, url) {
        $.ajax({   
            url: "https://www.thejakartapost.com/account/is-bookmarked", 
            data: {
                url: url,
            }, 
            success: function(response) {
                if (response.result) {
                    $('.addToBookmark-' + i).parent().attr("bookmark", "true");
                    $('.addToBookmark-' + i).parent().find('h5').html("Bookmarked");
                    $('.addToBookmark-' + i).parent().find('span').html("Bookmarked");
                } else {
                    $('.addToBookmark-' + i).parent().attr("bookmark", "false");
                }
            },
            error: function(){
                $(".show-share").html('0');
            }
        });
    }

    function getShareCount(url_current, title_current) {
        $.ajax(
            {   
                url: 'https://shareapi.thejakartapost.com/countshare/', 
                data: {
                        url: url_current,
                        title: title_current,
                    }, 
                success: function(result) {
                    objshare = $('.show-share:not([data-href])');
                    objshare.attr('data-href', url_current);
                    objshare.html(result);

                },
                error: function(){
                    objshare = $('.show-share:not([data-href])');
                    objshare.attr('data-href', '-');
                    objshare.html(0);
                }   
            }
        );
    }
